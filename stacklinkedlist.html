public class Stack {

    Node top;
    
    public Stack() {
        top = null;
    }
    
    public boolean isEmpty() {
        return top == null;
    }
    
    public void push(int data) {
        Node newnode = new Node(data);
        
        if (isEmpty()) {
            top = newnode;
        } else {
            newnode.next = top;
            top = newnode;    
        }
    }
    
    public int pop() {
        if (isEmpty()) {
            System.out.println("Stack is empty");
            return -1;
        }
        
        int popval = top.data;
        top = top.next;
        return popval;
    }
    
    public int peek() {
        return top.data;
    }
    
    public void display() {
        Node current = top;
        
        if (isEmpty()) {
            System.out.println("Stack is empty");
            return;
        }
        
        while (current != null) {
            System.out.println(current.data);
            current = current.next;
        }
    }
}
